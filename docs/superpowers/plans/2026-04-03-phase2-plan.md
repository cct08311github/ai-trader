```sql
-- research.db: sector_data (daily aggregated sector metrics)
CREATE TABLE IF NOT EXISTS sector_data (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    trade_date      TEXT    NOT NULL,           -- ISO-8601 YYYY-MM-DD
    sector_code     TEXT    NOT NULL,           -- TWSE sector code (e.g. '24' = 半導體業)
    sector_name     TEXT    NOT NULL,           -- 產業名稱
    market_cap      REAL,                       -- 總市值 (億元)
    turnover        REAL,                       -- 成交金額 (億元)
    change_pct      REAL,                       -- 指數漲跌幅 (%)
    fund_flow_net   REAL,                       -- 法人淨買賣超 (億元, aggregated from eod_institution_flows)
    pe_ratio        REAL,                       -- 產業平均本益比
    stock_count     INTEGER,                    -- 產業內上市股票數
    source          TEXT    NOT NULL DEFAULT 'twse',
    fetched_at      TEXT    NOT NULL DEFAULT (datetime('now')),
    UNIQUE (trade_date, sector_code)
);

-- research.db: sector_mapping (symbol to sector classification)
CREATE TABLE IF NOT EXISTS sector_mapping (
    symbol          TEXT    NOT NULL,           -- 股票代碼 (e.g. '2330')
    sector_code     TEXT    NOT NULL,
    sector_name     TEXT    NOT NULL,
    sub_sector      TEXT,                       -- 子產業（如有）
    updated_at      TEXT    NOT NULL DEFAULT (datetime('now')),
    PRIMARY KEY (symbol)
);

CREATE INDEX IF NOT EXISTS idx_sector_data_date    ON sector_data (trade_date DESC);
CREATE INDEX IF NOT EXISTS idx_sector_data_code    ON sector_data (sector_code);
CREATE INDEX IF NOT EXISTS idx_sector_mapping_code ON sector_mapping (sector_code);
```

### Implementation Details

**WebGL Detection:**
```javascript
function hasWebGL() {
  try {
    const canvas = document.createElement('canvas')
    return !!(canvas.getContext('webgl') || canvas.getContext('webgl2'))
  } catch {
    return false
  }
}

function isMobile() {
  return /Android|iPhone|iPad|iPod/i.test(navigator.userAgent)
    || window.innerWidth < 768
}

// Decision logic:
// - Mobile: always 2D (performance)
// - Desktop without WebGL: 2D fallback
// - Desktop with WebGL: 3D globe
const use3D = hasWebGL() && !isMobile()
```

#### Design

```python
# stream.py — new SSE event type
# Extends existing SSE infrastructure (same Cursor/Semaphore pattern)
#
# New endpoint: GET /api/stream/market-ticks
# Event format:
#   event: market_tick
#   data: {"symbol": "^TWII", "price": 18523.5, "change_pct": 0.52, "ts": "..."}
#
# Data source: in-memory ring buffer (collections.deque, maxlen=200)
# market_index_fetcher writes to buffer on each fetch cycle
# SSE reads from buffer, sends delta only (new entries since last cursor)
```

**Ring Buffer Architecture:**
```
market_index_fetcher ──write──→ [Ring Buffer (deque, 200)] ←──read── SSE stream
                                   │
                                   └── No DB writes for ticks (existing daily store unchanged)
```

**Frontend GlobalTicker Change:**
- Current: `useQuery` polling every 60s
- New: `EventSource('/api/stream/market-ticks')` + `onmessage` handler
- Fallback: if SSE disconnects, revert to polling

```javascript
// Client-side PDF generation — no backend changes needed
import html2canvas from 'html2canvas'
import { jsPDF } from 'jspdf'

async function exportReportPDF(elementId, filename) {
  const element = document.getElementById(elementId)
  const canvas = await html2canvas(element, {
    scale: 2,                    // retina quality
    useCORS: true,               // cross-origin images
    backgroundColor: '#0d131e',  // dark theme background
  })
  const imgData = canvas.toDataURL('image/png')
  const pdf = new jsPDF('p', 'mm', 'a4')
  const width = pdf.internal.pageSize.getWidth()
  const height = (canvas.height * width) / canvas.width
  pdf.addImage(imgData, 'PNG', 0, 0, width, height)
  pdf.save(filename)
}
```
